Primeira chamada em 5 minutos

Do zero à primeira simulação de Capital de Giro — sem configuração manual, sem aprovação humana.

✅
Sandbox self-service. Credenciais geradas instantaneamente ao criar sua conta. Nenhum dado real é processado.

Passo a passo

1
Crie sua conta e gere credenciais
Clique em Gerar credenciais no topo. O wizard gera client_id e client_secret instantaneamente.
2
Obtenha o Bearer Token
Use o fluxo Client Credentials do OAuth 2.0.
# POST para obter o token
curl -X POST https://auth.bradesco.com.br/oauth2/token \
  -H "Content-Type: application/x-www-form-urlencoded" \
  -d "grant_type=client_credentials" \
  -d "client_id=SEU_CLIENT_ID" \
  -d "client_secret=SEU_CLIENT_SECRET" \
  -d "scope=caas:read caas:write"
3
Simule um Capital de Giro
Calcule condições de crédito para um cliente PJ de R$ 50.000.
curl -X POST https://sandbox.bradesco.com.br/v1/caas/simulacao/grade \
  -H "Authorization: Bearer SEU_TOKEN" \
  -H "Content-Type: application/json" \
  -H "IdempotencyKey: $(uuidgen)" \
  -d '{
    "vlrSolicitado": 50000.00,
    "percJurosNegociado": 2.5,
    "tipoPessoa": 2
  }'

🪝 CaaS

Webhooks

Notificações assíncronas de mudança de status. Configure sua URL HTTPS no Console — sem aprovação manual.

✅
Bradesco tenta a entrega até 5 vezes com backoff exponencial (1s → 2s → 4s → 8s → 16s). Após 5 falhas, o evento vai para a dead-letter queue e você recebe um alerta por e-mail.

Eventos disponíveis

EventoQuando é disparadoPayload principal
proposta.criadaNova proposta incluídapropostaId, status
proposta.assinadaCCB assinada pelo clientepropostaId, dtAssinatura

proposta.liberadaNa fila de pagamento BMPpropostaId, dtLiberacao
desembolso.realizadoPagamento confirmadopropostaId, vlrDesembolsado, dtPagamento
proposta.canceladaCancelada por qualquer partepropostaId, motivoCancelamento

Validação de assinatura HMAC-SHA256

Cada requisição inclui o header X-Bradesco-Signature: sha256=<hash>. Valide sempre antes de processar o evento.

// Express.js — validação de webhook Bradesco
const crypto = require('crypto');
const express = require('express');
const app = express();

// Use raw body para cálculo de HMAC
app.use(express.raw({ type: 'application/json' }));

function validateSignature(rawBody, signature, secret) {
  const expected = 'sha256=' + crypto
    .createHmac('sha256', secret)
    .update(rawBody)
    .digest('hex');
  return crypto.timingSafeEqual(
    Buffer.from(expected, 'utf8'),
    Buffer.from(signature, 'utf8')
  );
}

app.post('/webhooks/bradesco', (req, res) => {
  const sig = req.headers['x-bradesco-signature'];
  const secret = process.env.BRADESCO_WEBHOOK_SECRET;

  if (!validateSignature(req.body, sig, secret)) {
    return res.status(401).json({ error: 'Assinatura inválida' });
  }

  const evento = JSON.parse(req.body);
  switch (evento.evento) {
    case 'desembolso.realizado':
      console.log(`Desembolso: R$ ${evento.dados.vlrDesembolsado}`);
      // processar desembolso...
      break;
    case 'proposta.cancelada':
      // notificar cliente...
      break;
  }
  res.status(200).json({ received: true });
});
# FastAPI — validação de webhook Bradesco
import hmac, hashlib, os
from fastapi import FastAPI, Request, HTTPException
from fastapi.responses import JSONResponse

app = FastAPI()

def validate_signature(raw_body: bytes, signature: str, secret: str) -> bool:
    expected = "sha256=" + hmac.new(
        secret.encode("utf-8"),
        raw_body,
        hashlib.sha256
    ).hexdigest()
    # comparação segura contra timing attacks
    return hmac.compare_digest(expected, signature)

@app.post("/webhooks/bradesco")
async def handle_webhook(request: Request):
    raw_body = await request.body()
    signature = request.headers.get("x-bradesco-signature", "")
    secret = os.environ["BRADESCO_WEBHOOK_SECRET"]

    if not validate_signature(raw_body, signature, secret):
        raise HTTPException(status_code=401, detail="Assinatura inválida")

    evento = await request.json()
    tipo = evento.get("evento")

    if tipo == "desembolso.realizado":
        valor = evento["dados"]["vlrDesembolsado"]
        print(f"✅ Desembolso realizado: R$ {valor:,.2f}")
        # processar notificação ao parceiro...

    elif tipo == "proposta.assinada":
        proposta_id = evento["dados"]["propostaId"]
        print(f"✍️  CCB assinada: {proposta_id}")
        # acionar liberação automática...

    return JSONResponse({"received": True})
// Spring Boot — validação de webhook Bradesco
import org.springframework.web.bind.annotation.*;
import javax.crypto.Mac;
import javax.crypto.spec.SecretKeySpec;
import java.nio.charset.StandardCharsets;
import java.security.MessageDigest;
import java.util.HexFormat;

@RestController
@RequestMapping("/webhooks")
public class BradescoWebhookController {

    @Value("${bradesco.webhook.secret}")
    private String webhookSecret;

    @PostMapping("/bradesco")
    public ResponseEntity<?> handleWebhook(
        @RequestBody byte[] rawBody,
        @RequestHeader("X-Bradesco-Signature") String signature
    ) {
        if (!isValidSignature(rawBody, signature)) {
            return ResponseEntity.status(401)
                .body(Map.of("error", "Assinatura inválida"));
        }

        WebhookEvent evento = objectMapper.readValue(rawBody, WebhookEvent.class);

        switch (evento.getEvento()) {
            case "desembolso.realizado" ->
                desembolsoService.processar(evento.getDados());
            case "proposta.assinada" ->
                propostaService.marcarAssinada(evento.getDados());
            case "proposta.cancelada" ->
                notificacaoService.notificarCancelamento(evento.getDados());
        }
        return ResponseEntity.ok(Map.of("received", true));
    }

    private boolean isValidSignature(byte[] body, String signature) {
        try {
            Mac mac = Mac.getInstance("HmacSHA256");
            mac.init(new SecretKeySpec(
                webhookSecret.getBytes(StandardCharsets.UTF_8), "HmacSHA256"));
            String expected = "sha256=" +
                HexFormat.of().formatHex(mac.doFinal(body));
            // MessageDigest.isEqual é timing-safe
            return MessageDigest.isEqual(
                expected.getBytes(StandardCharsets.UTF_8),
                signature.getBytes(StandardCharsets.UTF_8));
        } catch (Exception e) { return false; }
    }
}

Cobranças PIX

Crie cobranças com QR Code dinâmico. O pagador escaneia com qualquer app bancário e o pagamento é liquidado em segundos.

POST /pix/cob — Criar cobrança imediata

Gera uma cobrança com QR Code e copia-e-cola. txid é gerado automaticamente se omitido.

curl -X POST https://sandbox.bradesco.com.br/v1/pix/cob \
  -H "Authorization: Bearer {token}" \
  -H "Content-Type: application/json" \
  -d '{
    "calendario": { "expiracao": 3600 },
    "valor": { "original": "150.00" },
    "chave": "bradesco@bradesco.com.br",
    "solicitacaoPagador": "Pagamento referente ao pedido #12345",
    "infoAdicionais": [
      { "nome": "Pedido", "valor": "12345" }
    ]
  }'

Validação de assinatura

O Bradesco inclui o header x-bradesco-signature com HMAC-SHA256 do payload. Valide sempre antes de processar.

// Express.js — validação de webhook PIX
const crypto = require('crypto');
app.use(express.raw({ type: '*/*' }));

app.post('/webhooks/pix', (req, res) => {
  const sig = req.headers['x-bradesco-signature'];
  const expected = 'sha256=' + crypto
    .createHmac('sha256', process.env.PIX_WEBHOOK_SECRET)
    .update(req.body).digest('hex');

  if (!crypto.timingSafeEqual(Buffer.from(expected), Buffer.from(sig)))
    return res.status(401).end();

  const { pix } = JSON.parse(req.body);
  pix.forEach(p => {
    console.log(`PIX recebido: R$ ${p.valor} de ${p.pagador.nome}`);
    // processar liquidação do pedido p.txid...
  });
  res.status(200).end();
});
# FastAPI — validação webhook PIX
import hmac, hashlib, os
from fastapi import FastAPI, Request, HTTPException

app = FastAPI()

@app.post("/webhooks/pix")
async def pix_webhook(request: Request):
    raw = await request.body()
    sig = request.headers.get("x-bradesco-signature", "")
    secret = os.environ["PIX_WEBHOOK_SECRET"]
    expected = "sha256=" + hmac.new(
        secret.encode(), raw, hashlib.sha256).hexdigest()
    if not hmac.compare_digest(expected, sig):
        raise HTTPException(status_code=401)
    data = await request.json()
    for pix in data["pix"]:
        print(f"PIX: R$ {pix['valor']} de {pix['pagador']['nome']}")
    return {"ok": True}
// Spring Boot — validação webhook PIX
@PostMapping("/webhooks/pix")
public ResponseEntity<?> pixWebhook(
    @RequestBody byte[] body,
    @RequestHeader("x-bradesco-signature") String sig) {
  Mac mac = Mac.getInstance("HmacSHA256");
  mac.init(new SecretKeySpec(
    webhookSecret.getBytes(StandardCharsets.UTF_8), "HmacSHA256"));
  String expected = "sha256=" +
    HexFormat.of().formatHex(mac.doFinal(body));
  if (!MessageDigest.isEqual(expected.getBytes(), sig.getBytes()))
    return ResponseEntity.status(401).build();
  PixWebhookPayload payload = mapper.readValue(body, PixWebhookPayload.class);
  payload.getPix().forEach(pixService::processarRecebimento);
  return ResponseEntity.ok().build();
}
⚠️
Responda HTTP 200 em até 5 segundos. Bradesco reexecuta até 3 vezes com backoff de 5min. Após 3 falhas, o evento vai para a dead-letter queue.
